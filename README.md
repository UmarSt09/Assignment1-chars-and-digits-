#Данный код считывает строку,
#затем количество бувк и цифр в ней



s=input() #сначало мы вводим строку, 
с которой будем считывать количество 


bukva=0 #создаем новую переменную, 
в которой будем считать количество букв

cyfra=0 #создаем новую переменную, 
в которой будем считать количество цифр

for i in s: #цикл, где мы
берем каждый символ строки отдельно 

    if i.isalpha: #Проверяем, если наш символ
		это буква, то добавляем +1 к счёту 
       
				bukva+=1 #Здесь ведём счёт букв
    
		elif i.isdigit: #Проверяем, если наш символ
		это цифра, то добавляем +1 к счёту
        
				cyfra+=1 #Здесь ведём счёт цифр 


print(f"В строке:{bukva} буквы и {cyfra} цифры")
#теперь выводим всё с помощью "print",
используем f"", для свободного ввода 
строк и переменных в круглых скобках {}
